function haversineDistanceKm(a: { lat: number; lng: number }, b: { lat: number; lng: number }) {
  const R = 6371
  const toRad = (x: number) => (x * Math.PI) / 180
  const dLat = toRad(b.lat - a.lat)
  const dLon = toRad(b.lng - a.lng)
  const lat1 = toRad(a.lat)
  const lat2 = toRad(b.lat)
  const h =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.sin(dLon / 2) * Math.sin(dLon / 2) * Math.cos(lat1) * Math.cos(lat2)
  const c = 2 * Math.atan2(Math.sqrt(h), Math.sqrt(1 - h))
  return R * c
}

function haversineDistanceKm(a: { lat: number; lng: number }, b: { lat: number; lng: number }) {
  const R = 6371
  const toRad = (x: number) => (x * Math.PI) / 180
  const dLat = toRad(b.lat - a.lat)
  const dLon = toRad(b.lng - a.lng)
  const lat1 = toRad(a.lat)
  const lat2 = toRad(b.lat)
  const h =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.sin(dLon / 2) * Math.sin(dLon / 2) * Math.cos(lat1) * Math.cos(lat2)
  const c = 2 * Math.atan2(Math.sqrt(h), Math.sqrt(1 - h))
  return R * c
}

                  </span>
                )}
              </div>
            )}
          </div>
          <div className="text-right ml-4">
            <p className="text-sm text-gray-400">Fee</p>
            <p className="text-xl font-semibold text-white">‚Çπ{match.fee.toLocaleString()}</p>
          </div>
        </div>
      </div>
    )
  }

  // New state for course names and suggestions
  const [courseNames, setCourseNames] = useState<string[]>([])
  const [courseSuggestions, setCourseSuggestions] = useState<string[]>([])
  const [showCourseDropdown, setShowCourseDropdown] = useState(false)

  // Fetch course names on mount
  useEffect(() => {
    api.get("/courses/course-names")
      .then(res => {
        if (Array.isArray(res.data?.courses)) {
          setCourseNames(res.data.courses)
        }
      })
      .catch(() => setCourseNames([]))
  }, [])

  // Filter suggestions on keystroke
  useEffect(() => {
    const q = courseQuery.trim().toLowerCase()
    if (q.length < 1) {
      setCourseSuggestions([])
      setShowCourseDropdown(false)
      return
    }
    const matches = courseNames.filter(name =>
      name.toLowerCase().includes(q)
    )
    setCourseSuggestions(matches)
    setShowCourseDropdown(matches.length > 0)
  }, [courseQuery, courseNames])

  // Handle suggestion click
  const selectCourseSuggestion = (name: string) => {
    setCourseQuery(name)
    setCourseSuggestions([])
    setShowCourseDropdown(false)
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-black via-gray-900 to-black text-white py-20 px-6">
      <div className="max-w-6xl mx-auto">
        <h1 className="text-3xl md:text-4xl font-bold text-center text-cyan-400 mb-10">Find Institutions</h1>

        {/* Error message above the form */}
        {formError && (
          <div className="mb-4 p-4 rounded-lg bg-red-500/10 border border-red-500/30 text-red-400 text-center">
            {formError}
          </div>
        )}

        {/* Unified Search Form */}
        <motion.form
          onSubmit={handleUnifiedSearch}
          initial={{ opacity: 0, y: 10 }}
          animate={{ opacity: 1, y: 0 }}
          className="bg-gray-900/60 border border-cyan-500/20 rounded-2xl p-6"
        >
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            {/* Row 1: Course Name | Location Dropdown | Location Input */}
            <div className="flex flex-col md:flex-row gap-4 md:col-span-2">
              {/* Course Name */}
              <div style={{ position: "relative" }} className="flex-1">
                <label className="block text-sm text-gray-300 mb-1">Course Name</label>
                <input
                  type="text"
                  value={courseQuery}
                  onChange={e => setCourseQuery(e.target.value)}
                  className="w-full px-4 py-3 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
                  placeholder="e.g. React, Data Science"
                  autoComplete="off"
                  onFocus={() => {
                    if (courseSuggestions.length > 0) setShowCourseDropdown(true)
                  }}
                  onBlur={e => {
                    setTimeout(() => setShowCourseDropdown(false), 200)
                  }}
                  required
                />
                {showCourseDropdown && (
                  <div
                    className="absolute left-0 right-0 rounded-lg border border-gray-700 bg-gray-900/80 max-h-56 overflow-auto z-10"
                    style={{ top: "100%" }}
                  >
                    {courseSuggestions.map((name, i) => (
                      <button
                        key={i}
                        type="button"
                        onMouseDown={e => {
                          e.preventDefault()
                          selectCourseSuggestion(name)
                        }}
                        className="w-full text-left px-3 py-2 hover:bg-gray-800 text-sm text-gray-200"
                      >
                        {name}
                      </button>
                    ))}
                  </div>
                )}
              </div>
              {/* Location Dropdown */}
              <div className="flex-1 min-w-[140px]">
                <label className="block text-sm text-gray-300 mb-1">Location Type</label>
                <select
                  value={locationMode}
                  onChange={e => setLocationMode(e.target.value as "manual" | "nearest")}
                  className="w-full px-2 py-3 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
                  required
                >
                  <option value="manual">Preferred Location</option>
                  <option value="nearest">Find Nearest</option>
                </select>
              </div>
              {/* Location Input */}
              <div className="flex-1" style={{ position: "relative" }}>
                <label className="block text-sm text-gray-300 mb-1">Location</label>
                {locationMode === "nearest" ? (
                  <button
                    type="button"
                    onClick={requestLocation}
                    className="w-full px-3 py-3 rounded-lg bg-gray-800/60 border border-gray-600 text-white hover:bg-gray-800 transition flex items-center gap-2"
                    disabled={locLoading}
                  >
                    {locLoading ? (
                      <span className="flex items-center gap-2">
                        <span>Accessing</span>
                        <span className="w-4 h-4 border-2 border-gray-400 border-t-cyan-400 rounded-full animate-spin"></span>
                      </span>
                    ) : userCoords ? (
                      "Location Set ‚úì"
                    ) : (
                      "Allow Location"
                    )}
                  </button>
                ) : (
                  <>
                    <input
                      type="text"
                      value={locationQuery}
                      onChange={e => setLocationQuery(e.target.value)}
                      className="w-full px-4 py-3 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
                      placeholder="e.g. Mumbai, Delhi"
                      required
                    />
                    {locationSuggestions.length > 0 && (
                      <div
                        className="absolute left-0 right-0 rounded-lg border border-gray-700 bg-gray-900/80 max-h-56 overflow-auto z-10"
                        style={{ top: "100%" }}
                      >
                        {locationSuggestions.map((s, i) => (
                          <button
                            key={i}
                            type="button"
                            onClick={() => selectLocationSuggestion(s)}
                            className="w-full text-left px-3 py-2 hover:bg-gray-800 text-sm text-gray-200"
                          >
                            {s}
                          </button>
                        ))}
                        {locLoading && (
                          <div className="px-3 py-2 text-xs text-gray-400">Loading‚Ä¶</div>
                        )}
                      </div>
                    )}
                  </>
                )}
              </div>
            </div>

            {/* Row 2: Max Fee | Min Fee */}
            <div>
              <label className="block text-sm text-gray-300 mb-1">Max Price (‚Çπ)</label>
              <input
                type="number"
                value={prefFilters.priceMax ?? ""}
                onChange={e => setPrefFiltersMemo({ ...prefFilters, priceMax: e.target.value ? Number(e.target.value) : undefined })}
                className="w-full px-3 py-2 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
                placeholder="e.g. 60000"
                required
              />
            </div>
            <div>
              <label className="block text-sm text-gray-300 mb-1">Min Price (‚Çπ)</label>
              <input
                type="number"
                value={prefFilters.priceMin ?? ""}
                onChange={e => setPrefFiltersMemo({ ...prefFilters, priceMin: e.target.value ? Number(e.target.value) : undefined })}
                className="w-full px-3 py-2 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
                placeholder="e.g. 20000"
                required
              />
            </div>

            {/* Row 3: Duration | Sort */}
            <div>
              <label className="block text-sm text-gray-300 mb-1">Max Duration (months)</label>
              <select
                value={prefFilters.durationMax ?? ""}
                onChange={e => setPrefFiltersMemo({ ...prefFilters, durationMax: e.target.value ? Number(e.target.value) : undefined })}
                className="w-full px-3 py-2 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
                required
              >
                <option value="">Any</option>
                <option value="3">Up to 3</option>
                <option value="6">Up to 6</option>
                <option value="9">Up to 9</option>
                <option value="12">Up to 12</option>
                <option value="24">Up to 24</option>
              </select>
            </div>
            <div>
              <label className="block text-sm text-gray-300 mb-1">Sort by Price</label>
              <select
                value={prefFilters.sortPrice ?? ""}
                onChange={e => setPrefFiltersMemo({ ...prefFilters, sortPrice: (e.target.value as any) })}
                className="w-full px-3 py-2 rounded-lg bg-black/60 border border-gray-600 text-white focus:ring-2 focus:ring-cyan-400 outline-none"
              >
                <option value="">None</option>
                <option value="asc">Low to High</option>
                <option value="desc">High to Low</option>
              </select>
            </div>

            {/* Submit button spanning both columns */}
            <div className="md:col-span-2">
              <motion.button
                whileHover={{ scale: 1.02 }}
                whileTap={{ scale: 0.98 }}
                type="submit"
                className="w-full py-3 rounded-lg bg-cyan-500 text-black font-semibold"
              >
                Search
              </motion.button>
            </div>
          </div>
        </motion.form>

        {/* Results */}
        <div className="mt-4 grid grid-cols-1 md:grid-cols-2 gap-4">
          {prefLoading && (
            <p className="text-gray-400">Searching...</p>
          )}
          {!prefLoading && prefMessage && (
            <div className="col-span-full mb-4 p-4 rounded-lg bg-blue-500/10 border border-blue-500/30 text-blue-400 text-center">
              <p className="text-sm">{prefMessage}</p>
            </div>
          )}
          {!prefLoading && prefError && (
            <p className="text-yellow-400 col-span-full">{prefError}</p>
          )}
          {prefSearched && prefResults.length === 0 && !prefMessage && (
            <p className="text-gray-400 col-span-full">No institutions match your search.</p>
          )}
          {prefResults.map(match => (
            <CourseMatchCard key={match.id} match={match} />
          ))}
        </div>
      </div>
    </div>
  )
}
